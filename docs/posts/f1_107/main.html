# Create the cache directory if it doesn't exist
cache_dir = 'f1_cache'

def fetch_and_save_f1_data_with_telemetry(year, output_dir='f1_data'):
    # Ensure output directory exists
    if not os.path.exists(output_dir):
        os.makedirs(output_dir)

    # Fetch the event schedule for the year
    schedule = ff1.get_event_schedule(year)

    # Iterate over each event in the schedule
    for event in schedule.itertuples():
        event_name = event.EventName.replace(' ', '_')  # Format event name for folder
        event_folder = os.path.join(output_dir, f"{year}_{event_name}")

        # Create a folder for the event
        if not os.path.exists(event_folder):
            os.makedirs(event_folder)

        print(f"Processing Event: {event.EventName} (Round {event.RoundNumber})")

        # List of possible session types
        session_types = ['FP1', 'FP2', 'FP3', 'Q', 'Sprint', 'Race']

        for session_type in session_types:
            try:
                # Load the session
                session = ff1.get_session(year, event.RoundNumber, session_type)
                session.load()
                print(f"  Loaded {session_type} session.")

                # Save session data
                # 1. Lap Data
                lap_data = session.laps
                if not lap_data.empty:
                    lap_filename = os.path.join(event_folder, f"{session_type}_laps.csv")
                    lap_data.to_csv(lap_filename, index=False)
                    print(f"    Saved lap data to {lap_filename}")

                # 2. Weather Data
                weather_data = session.weather_data
                if not weather_data.empty:
                    weather_filename = os.path.join(event_folder, f"{session_type}_weather.csv")
                    weather_data.reset_index().to_csv(weather_filename, index=False)
                    print(f"    Saved weather data to {weather_filename}")

                # 3. Driver Timing Data
                if session.results is not None:
                    results_df = session.results.reset_index()
                    results_filename = os.path.join(event_folder, f"{session_type}_results.csv")
                    results_df.to_csv(results_filename, index=False)
                    print(f"    Saved results data to {results_filename}")

                # 4. Telemetry Data (Driver-Specific)
                drivers = session.drivers
                for driver in drivers:
                    try:
                        driver_name = session.get_driver(driver)['Abbreviation']
                        telemetry = session.car_data[driver].add_distance()  # Add distance column
                        if not telemetry.empty:
                            telemetry_filename = os.path.join(
                                event_folder,
                                f"{session_type}_telemetry_{driver_name}.csv"
                            )
                            telemetry.to_csv(telemetry_filename, index=False)
                            print(f"    Saved telemetry data for {driver_name} to {telemetry_filename}")
                    except Exception as te:
                        print(f"      Failed to fetch telemetry data for driver {driver}: {te}")

            except Exception as e:
                print(f"  Failed to load {session_type} session for {event.EventName}: {e}")
